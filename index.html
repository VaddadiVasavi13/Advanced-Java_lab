Week 1: Introduction to Advanced Java

What are the differences between Core Java and Advanced Java?
Core Java includes basics like syntax, datatypes, classes, objects, inheritance, and threading concepts, while Advanced Java covers Applets, Servlets, frameworks, how to connect the program & database, and networking.

What are the main important features of Applets in Java?
Java Applets were small programs that run inside a web browser. Also known as Standalone Applications.
‚Ä¢ Small and lightweight.

What are the key features of Java Servlets?
Servlets extend the functionality of a web server by handling requests and generating dynamic responses. They work on the Server side, have better performance, integrate easily with databases, and are also known as Web Applications.

What is the difference between Applets and Servlets in Java?
Applets
‚Ä¢ Executed on Client side.
‚Ä¢ Uses UI classes like AWT & Swing.
‚Ä¢ Requires a Java-compatible browser for execution.

Servlets
‚Ä¢ Executed on Server side.
‚Ä¢ No user interface is required.
‚Ä¢ Accepts input from browser and generates response as an HTML page.

What are the major features of Java compared to other programming languages?
Platform independence, object-oriented programming, robustness and security, and multithreading.

What is the main role of JDBC and ODBC in database connectivity?
JDBC (Java Database Connectivity): Java API to connect and execute queries with databases.
ODBC (Open Database Connectivity): Language-independent API for database access.
JDBC provides a way for Java applications to interact with a database using SQL.

What are the differences between ODBC and JDBC?
JDBC is designed for Java applications and interacts easily with relational databases, while ODBC is language-agnostic and can be used with various programming languages.

What are the differences between Cookies and Sessions in web applications?
Cookies are small pieces of data stored in the user's browser and help remember things like login status or preferences.
Sessions store user data on the server, making them more secure and suitable for temporary or sensitive data.

What are the differences between Spring Boot and the traditional Spring Framework?
Spring Framework provides the foundation and flexibility for building complex applications, while Spring Boot simplifies and accelerates the development process.

What are the differences between standalone applications and web applications?
Web apps are used through a browser, while standalone apps (desktop/offline apps) can be accessed offline.

What are the basic requirements for executing a web application?
Database, Browser, Backend, Frontend, Input Validation, JavaScript, and Web Server.

package callable_stament;

import java.sql.*;
import java.util.Scanner;

public class Callable_stament {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter Book ID: ");
        int bookId = scanner.nextInt();
     
        String url = "jdbc:mysql://localhost/procedures"; 
        String user = "root";
        String password = "";
        Connection conn = null;
        CallableStatement cstmt = null;

        try {
            Class.forName("com.mysql.jdbc.Driver");

            conn = DriverManager.getConnection(url, user, password);
            String sql = "{CALL getBookInfo1(?, ?, ?, ?, ?)}";
            cstmt = conn.prepareCall(sql);
            cstmt.setInt(1, bookId);
            cstmt.registerOutParameter(2, Types.VARCHAR); 
            cstmt.registerOutParameter(3, Types.VARCHAR);  
            cstmt.registerOutParameter(4, Types.DECIMAL); 
            cstmt.registerOutParameter(5, Types.INTEGER);  
            
            cstmt.execute();
            String title = cstmt.getString(2);
            String author = cstmt.getString(3);
            double price = cstmt.getDouble(4);
            int stock = cstmt.getInt(5);
            if (title != null) {
                System.out.println("Book Details:");
                System.out.println("Title: " + title);
                System.out.println("Author: " + author);
                System.out.println("Price: $" + price);
                System.out.println("Available Stock: " + stock);
            } else {
                System.out.println("No book found with ID " + bookId);
            }
        } catch (ClassNotFoundException e) {
            System.out.println("MySQL JDBC Driver not found.");
            e.printStackTrace();
        } catch (SQLException e) {
            System.out.println("Database error:");
            e.printStackTrace();
        } finally {
            try {
                if (cstmt != null) cstmt.close();
                if (conn != null) conn.close();
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
            scanner.close();
        }
    }
}

    FROM books
    WHERE book_id = bookId;

Week 5: JDBC ResultSet

Aim:

To demonstrate JDBC ResultSet navigation using a scroll-insensitive, read-only ResultSet in a MySQL database connection.

Code:

package result_set;
import java.sql.*;

public class Result_set {
    public static void main(String[] args) {
        String jdbcURL = "jdbc:mysql://localhost/employee";
        String username = "root";
        String password = "";

        try {
            Class.forName("com.mysql.jdbc.Driver");
            Connection connection = DriverManager.getConnection(jdbcURL, username, password);

            // Create a scrollable, read-only Statement
            Statement statement = connection.createStatement(
                ResultSet.TYPE_SCROLL_INSENSITIVE, // scrollable
                ResultSet.CONCUR_READ_ONLY          // read-only
            );

            ResultSet result = statement.executeQuery("SELECT eno, name FROM employee");

            // Move cursor to last row
            if (result.last()) {
                System.out.println("Last row - Employee No: " + result.getInt("eno") + ", Name: " + result.getString("name"));
            }

            // Move cursor to first row
            if (result.first()) {
                System.out.println("First row - Employee No: " + result.getInt("eno") + ", Name: " + result.getString("name"));
            }

            // Move cursor before the first row and then call next()
            result.beforeFirst();
            System.out.println("Iterating forward:");
            while (result.next()) {
                System.out.println("Employee No: " + result.getInt("eno") + ", Name: " + result.getString("name"));
            }

            // Jump to absolute row 3
            if (result.absolute(3)) {
                System.out.println("Absolute row 3 - Employee No: " + result.getInt("eno") + ", Name: " + result.getString("name"));
            }

            connection.close();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

AuthorService.java
package com.example.demo;

import java.util.ArrayList;
import java.util.List;
import org.slf4j.*;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Service;
import jakarta.annotation.PostConstruct;

@Service
public class AuthorService {
    private static final Logger log = LoggerFactory.getLogger(AuthorService.class);

    @Autowired
    JdbcTemplate jdbcTemplate;

    @PostConstruct
    public void postConstruct() {
        Author author1 = new Author("ramesh", "it");
        Author author2 = new Author("harshith", "it");
        List<Author> authors = new ArrayList<>();
        authors.add(author1);
        authors.add(author2);

        log.info("Creating tables");
        jdbcTemplate.execute("DROP TABLE author IF EXISTS");
        jdbcTemplate.execute("CREATE TABLE author(first_name varchar(255), last_name varchar(255))");
        authors.forEach(i -> jdbcTemplate.update("INSERT INTO author(first_name, last_name) VALUES (?,?)", i.getFirstName(), i.getLastName()));
        log.info("Records Saved");

        // retrieve saved records.
        log.info("Retrieving records");
        authors = jdbcTemplate.query("select * from author", (rs, rowNum) -> new Author(rs.getString("first_name"), rs.getString("last_name")));
        authors.forEach(i -> log.info(i.getFirstName() + " " + i.getLastName()));
    }
}

        model.addAttribute("employees", repo.findAll());
        return "employees";  // maps to src/main/resources/templates/employees.html
    }
}
